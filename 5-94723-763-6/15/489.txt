// Чисто виртуальные деструкторы требуют
// наличия тела функции
#include <iostream>
using namespace std;

class Pet {
public:
  virtual ~Pet() = 0;
};

Pet::~Pet() {
  cout << "~Pet()" << endl;
}

class Dog : public Pet {
public:
  ~Dog() {
    cout << "~Dog()" << endl;
  }
};

int main() {
  Pet* p = new Dog; // Повышающее приведение типа
  delete p; // Вызов виртуального деструктора
} ///:~
