#include "../require.h"
#include <cstring>

class Stringc {
string s;
public:
Stringc(const string& str = "") : s(str) {}
int strcmp(const Stringc& S) const {
return ::strcmp(s.c_str(), S.s.c_str());
}
// ... и т. д. для всех функций в string.h
};
