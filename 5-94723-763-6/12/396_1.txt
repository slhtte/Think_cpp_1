//: C12:TypeConversionAmbiguity.cpp
class Orange; // Class declaration

class Apple {
public:
operator Orange() const; // Приведение Apple к Orange
};

class Orange {
public:
Orange(Apple); // Приведение Apple к Orange
};

void f(Orange) {}

int main() {
Apple a;
//! f(a); // Ошибка: неоднозначность при приведении типа
} ///:~
