// Многократное использование кода посредством композиции
#include "Useful.h"

class Y {
  int i;
public:
  X x; // Внутренний объект
  Y() { i = 0; }
  void f(int ii) { i = ii; }
  int g() const { return i; }
};

int main() {
  Y y;
  y.f(47);
  y.x.set(37); // Обращение к внутреннему объекту
} ///:~
