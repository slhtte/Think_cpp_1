// Простые примеры использования ссылок в C++

int* f(int* x) {
  (*x)++;
  return x; // Безопасно, x существует вне функции
}

int& g(int& x) {
  x++; // Тот же эффект, что и в f()
  return x; // Безопасно, x существует вне функции
}

int& h() {
  int q;
//!  return q;  // Ошибка
  static int x;
  return x; // Безопасно, x существует вне функции
}

int main() {
  int a = 0;
  f(&a); // Некрасиво (но предельно ясно)
  g(a);  // Красиво (но смысл остается скрытым)
} ///:~
